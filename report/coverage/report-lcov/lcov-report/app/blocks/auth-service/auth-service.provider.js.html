(function() {
  'use strict';
 
  angular
    .module('blocks.auth-service')
    .provider('authService', authServiceProvider);
 
    authServiceProvider.$inject = ['$stateProvider'];
 
    /* @ngInject */
    function authServiceProvider($stateProvider) {
      var config = {};
 
      this.configure = function(cfg) {
        angular.extend(config, cfg);
      };
 
      this.$get = AuthService;
 
      AuthService.$inject = ['$rootScope', 'firebaseDataService'];
 
      /* @ngInject */
      function AuthService($rootScope, firebaseDataService) {
        var signedInUser = void 0;
        var service = {
          isUserSignedIn: isUserSignedIn
        }
 
        init();
        return service;
 
        //////////
 
        function init() {
          firebaseDataService
            .getCurrentUser()
            .then(function(user) {
              if (user) {
                signedInUser = user;
              } else {
                throw new Error('User data came empty');
              }
            })
            .catch(function(e) {
              console.log(e);
              signedInUser = void 0;
            })
            .finally(function() {
              console.log('Done checking user');
              $rootScope.$broadcast('authStateUpdate', signedInUser);
            });
 
        }
 
        function isUserSignedIn() {
          return signedInUser ? true : false;
        }
      }
    }
}());
